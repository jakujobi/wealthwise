# Commands for running the server
run dev server:
    python manage.py runserver

# Commands for creating an app
create app:
    python manage.py startapp [app name]

# Commands for database migrations
create databases for libraries:
    python manage.py migrate

create file to make models for database:
    python manage.py makemigrations [app name]

# Command for creating an admin user
create admin user:
    python manage.py createsuperuser

# Command to auto install required python libraries
install requirements:
    pip install -r requirements.txt

# Docker commands
docker-compose up

# Real-Time Messaging Commands
# Start Redis via Docker
docker run -d --name redis -p 6379:6379 redis:latest

# Start ASGI server with Daphne
daphne -b 127.0.0.1 -p 8000 core.asgi:application

# Or start with Uvicorn
uvicorn core.asgi:application --reload --host 127.0.0.1 --port 8000

# Celery worker for notifications
celery -A core worker -l info
